70 /*
71  * \brief perform plain BFS(breadth-first-search) (iterative) and push elements into queue
72  * \param v vertex to perform dfs call
73  * This function looks for an S-T path in residual graph. It also stores the parent of all the nodes visited along
74  * with the edge weights in order to enable ease of finding bottleneck edge weight while backtracking once
75  * destination is reached while searching(BFS)
76  */
77 int BFS(vector<pair<int,int>>&parent, Graph& residual, int src, int dest)
78 {
79  int v;
80  queue<int> q;
81  init(parent);
82  q.push(src);
83  parent[src]={ROOT,INT_MAX};
84  while(!q.empty())
85  {
86  v=q.front();
87  q.pop();
88 
89  if(v == dest)
90  {
91  /* If destination is reached, backtrack and check if parent to current node's
92  * residual capacity (stored in parent[child].second is the bottleneck value of S-T path.
93  */
94  int i=dest;
95  int bottleneck=INT_MAX;
96  while(parent[i].first!=ROOT)
97  {
98  if(abs(parent[i].second)<bottleneck)
99  {
100  bottleneck=abs(parent[i].second);
101  }
102  i=parent[i].first;
103  }
104  return bottleneck;
105  }
106  /* Since residual capacities of forward edges are stored in capacity vector and that of backward edges are
107  * stored in flow, we separately look for next edges in these 2 vectors in BFS.
108  * Further, note that flow (i.e backward edge) will be chosen only if no forward edge is available with
109  * positive capacity. To differentiate backward edge from forward in parent vector, parent[child].second
110  * is made negative for backward edges and positive for forward edges. Hence we use abs() function wherever necessary.
111  */
112  for(auto w:residual.vertex[v].capacity) //searching for a forward edge from current node to an unvisited node
113  {
114  if(w.second!=0 and parent[w.first].first==UNVISITED) //w.second has residual capacity from v to w.first
115  {
116  q.push(w.first);
117  parent[w.first].first=v;
118  parent[w.first].second=w.second; //positive
119  }
120  }
121  for(auto w:residual.vertex[v].flow) //searching for a backward edge from current node to an unvisited node
122  {
123  if(w.second!=0 and parent[w.first].first==UNVISITED)
124  {
125  q.push(w.first);
126  parent[w.first].first=v;
127  parent[w.first].second=(-w.second);//made negative.
128  /* Note: it was made negative just for ease of coding in augment function, where this distinction
129  * is essential. The residual capcity is actually and always a positive value(non-negative).
130  */
131  }
132  }
133  }
134  return 0;
135 }

170 /*
171  * \brief run the augment routine as explained in the class (and textbook).
172  * \param parent a vector for S-T path, original Graph, residual Graph, bottleneck and destination index
173  * This is where having negative edge weights comes handy. In this method bottleneck is added to forward edges
174  * and subtracted from reverse of backward edges in original graph. Note that parent was identified in residual
175  * graph while these updates are done in original graph. This is possible because both original and residual graphs
176  * have same vertex set. So j'th neighbour of i'th vertex of original will be same as that of residual(obv weights
177  * will be different).
178  */
179 void augment(vector<pair<int,int>>& parent, Graph& residual, Graph& original, int bottleneck, int dest)
180 {
181  int p=dest;
182  while(parent[p].first!=ROOT)
183  {
184  if(parent[p].second>0) //forward edge since positive weight value
185  {
186  for(int i=0;i<original.vertex[parent[p].first].flow.size();++i)
187  {
188  if(original.vertex[parent[p].first].flow[i].first==p) // note forward edge from parent[p] to p
189  {
190  original.vertex[parent[p].first].flow[i].second+=bottleneck;
191  break;
192  }
193  }
194  }
195  else //backward edge since weight with 0 will not be chosen in BFS hence weight must be -ve
196  {
197  for(int i=0;i<original.vertex[p].flow.size();++i)
198  {
199  if(original.vertex[p].flow[i].first==parent[p].first) //note reverse of backward edge hence p to parent[p]
200  {
201  original.vertex[p].flow[i].second-=bottleneck;
202  break;
203  }
204  }
205  }
206  p=parent[p].first;
207  }
208 }
